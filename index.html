const CACHE = 'lumen-v1'; const SHELL = [ '/', '/index.html', '/css/styles.css', '/js/app.js', '/js/api.js', '/js/store.js', '/js/ui.js', '/js/editor.js', '/icons/logo.svg', '/icons/icon-192.png', '/manifest.webmanifest', ]; self.addEventListener('install', (e) => { e.waitUntil(caches.open(CACHE).then((c) => c.addAll(SHELL)).then(() => self.skipWaiting())); }); self.addEventListener('activate', (e) => { e.waitUntil( caches.keys().then((keys) => Promise.all(keys.filter((k) => k !== CACHE).map((k) => caches.delete(k))) ).then(() => self.clients.claim()) ); }); self.addEventListener('fetch', (e) => { const { request } = e; if (request.method !== 'GET') return; const url = new URL(request.url); // Never cache API calls — always go to network. if (url.pathname.startsWith('/api/')) return; // App shell: cache-first, fall back to network, then offline index. e.respondWith( caches.match(request).then( (cached) => cached || fetch(request) .then((res) => { const copy = res.clone(); caches.open(CACHE).then((c) => c.put(request, copy)); return res; }) .catch(() => caches.match('/index.html')) ) ); });
